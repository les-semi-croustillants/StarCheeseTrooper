629

631  /* Changer la valeur pour ralentir ou acclrer le dplacement des fantomes */

894 void SDL_apply_surface( SDL_Surface* source, SDL_Surface* destination, int x, int y )
895 {
896  /* Make a temporary rectangle to hold the offsets */
897  SDL_Rect offset;
898 
899  /* Give the offsets to the rectangle */
900  offset.x = x;
